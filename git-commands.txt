# Git Commands for Repository Setup
# Copy and paste these commands one by one

# 1. Initialize git repository
git init

# 2. Add all files
git add .

# 3. Create first commit
git commit -m "Initial commit: Techmapperz website with auto-deployment setup"

# 4. Set main branch
git branch -M main

# 5. Add remote repository
git remote add origin https://github.com/Sahil-techmapperz/techmapperz-pord.git

# 6. Push to GitHub
git push -u origin main

# Alternative: If you already have a repository URL, use this:
git remote add origin https://github.com/Sahil-techmapperz/techmapperz-pord.git
git push -u origin main
